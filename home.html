See How Simple Fuzion Is

Clean, readable code that does exactly what you expect

web_server.fz 
# Create a simple web server in Fuzion
import web

# Define route handlers
home request:
    return "Welcome to Fuzion!"

api_users request:
    users = [
        {"name": "Alice", "age": 25},
        {"name": "Bob", "age": 30}
    ]
    return web.json(users)

# Start server
server = web.create_server port: 8080
server.route "/", home
server.route "/api/users", api_users
server.start

print "Server running at http://localhost:8080 üöÄ"
